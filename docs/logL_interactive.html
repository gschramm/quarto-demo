Log-likelihood landscape for a 2-voxel problem

This page shows the Poisson log-likelihood \(\mathcal{L}(x_1, x_2)\) for the 2-voxel forward model

\[\bar{y}_1 = x_1 + b_1, \qquad \bar{y}_2 = x_1 + x_2 + b_2.\]

Use the controls below to set the measured counts \((y_1, y_2)\), background levels \((b_1, b_2)\), axis range, colorbar range, and MLEM starting point. The plot updates automatically.

  • Blue triangle — constrained maximum \(\hat{\mathbf{x}}\) (non-negative orthant)
  • Red star — unconstrained maximum
  • Black square — MLEM starting point; black dots — 20 MLEM iterates

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.ticker import FormatStrFormatter

# Unpack form values (params arrives as a JavaScript object proxy)
y1       = int(params['y1'])
y2       = int(params['y2'])
b1       = float(params['b1'])
b2       = float(params['b2'])
xmin     = float(params['xmin'])
xmax     = float(params['xmax'])
lmin     = float(params['lmin'])
lmax     = float(params['lmax'])
x1_start = float(params['x1_start'])
x2_start = float(params['x2_start'])

# Guard against inverted ranges
xmin, xmax = min(xmin, xmax - 0.5), max(xmax, xmin + 0.5)
lmin, lmax = min(lmin, lmax - 1),   max(lmax, lmin + 1)

def logL(x1, x2, y1, y2, b1, b2):
    ybar1 = x1 + b1
    ybar2 = x1 + x2 + b2
    with np.errstate(divide="ignore", invalid="ignore"):
        term1 = np.where(
            y1 > 0,
            y1 * np.log(np.where(ybar1 > 0, ybar1, np.nan)) - ybar1,
            np.where(ybar1 > 0, -ybar1, -np.inf),
        )
        term2 = np.where(
            y2 > 0,
            y2 * np.log(np.where(ybar2 > 0, ybar2, np.nan)) - ybar2,
            np.where(ybar2 > 0, -ybar2, -np.inf),
        )
    return term1 + term2

def mlem_update(x1, x2, y1, y2, b1, b2):
    ybar1 = x1 + b1
    ybar2 = x1 + x2 + b2
    r1 = np.where(ybar1 > 0, y1 / ybar1, 0)
    r2 = np.where(ybar2 > 0, y2 / ybar2, 0)
    return float((x1 / 2) * (r1 + r2)), float(x2 * r2)

x = np.linspace(xmin, xmax, 800)
X1, X2 = np.meshgrid(x, x)
Z = logL(X1, X2, y1=y1, y2=y2, b1=b1, b2=b2)

# Show contours everywhere the log-likelihood is defined (both ȳ_i > 0)
defined  = (X1 + b1 > 0) & (X1 + X2 + b2 > 0)
Z_plot   = np.where(defined, Z, np.nan)
levels   = np.linspace(lmin, lmax, 40)

# Constrained maximum: grid search over the feasible (x >= 0) region
feasible     = (X1 >= 0) & (X2 >= 0)
Z_feasible   = np.where(feasible, Z_plot, np.nan)
flat_idx     = np.nanargmax(Z_feasible)
idx          = np.unravel_index(flat_idx, Z_feasible.shape)
xhat1, xhat2 = X1[idx], X2[idx]

# Unconstrained maximum
xml1 = y1 - b1
xml2 = y2 - xml1 - b2

# MLEM trajectory (20 steps from user-defined start point)
tx, ty = [x1_start], [x2_start]
for _ in range(20):
    nx, ny = mlem_update(tx[-1], ty[-1], y1, y2, b1, b2)
    tx.append(nx); ty.append(ny)

fig, ax = plt.subplots(figsize=(5.5, 4.5))
cs = ax.contourf(X1, X2, Z_plot, levels=levels, cmap="magma", alpha=0.85,
                 vmin=lmin, vmax=lmax)
ax.axhline(0, color="black", lw=1.0)
ax.axvline(0, color="black", lw=1.0)
ax.plot(tx,    ty,    "k.-", ms=5,  zorder=5, label="MLEM trajectory")
ax.plot(tx[0], ty[0], "ks",  ms=7,  zorder=6, label="MLEM start")
ax.plot(xhat1, xhat2, "b^",  ms=9,  zorder=6, label=r"$\hat{\mathbf{x}}$ (constrained)")
ax.plot(xml1,  xml2,  "r*",  ms=11, zorder=6, label="unconstrained max")
cb = plt.colorbar(cs, ax=ax, label=r"$\mathcal{L}(x_1,x_2)$",
                  fraction=0.046, pad=0.04)
cb.ax.yaxis.set_major_formatter(FormatStrFormatter("%.1f"))
ax.set_xlabel(r"$x_1$"); ax.set_ylabel(r"$x_2$")
ax.set_xlim(xmin, xmax); ax.set_ylim(xmin, xmax)
ax.set_aspect("equal", adjustable="box")
ax.set_title(
    rf"$y_1={y1},\; y_2={y2},\; b_1={b1:.1f},\; b_2={b2:.1f}$",
    fontsize="medium"
)
ax.legend(loc="upper right", fontsize=8)
plt.tight_layout()
plt.show()
